Database (Supabase):
├── profiles                     # User profiles table
│   ├── id (uuid, pk)           → References auth.users
│   ├── name (text)
│   ├── created_at (timestamp)
│   └── updated_at (timestamp)
├── books                        # Books table
│   ├── id (uuid, pk)
│   ├── title (text)
│   ├── author (text)
│   ├── description (text)
│   ├── genre (text)
│   ├── year (integer)
│   ├── added_by (uuid)         → References profiles(id)
│   ├── avg_rating (numeric)    # Auto-calculated
│   ├── reviews_count (integer) # Auto-calculated
│   ├── created_at (timestamp)
│   └── updated_at (timestamp)
└── reviews                      # Reviews table
    ├── id (uuid, pk)
    ├── book_id (uuid)          → References books(id)
    ├── user_id (uuid)          → References profiles(id)
    ├── rating (integer 1-5)
    ├── review_text (text)
    ├── created_at (timestamp)
    ├── updated_at (timestamp)
    └── UNIQUE(book_id, user_id) # One review per user per book
